Step 1: Create the Tables

CREATE TABLE users(
    user_id INT PRIMARY KEY,
    name VARCHAR(50),
    signup_date DATE
);

CREATE TABLE orders(
    order_id INT PRIMARY KEY,
    user_id INT,
    amount INT
);

CREATE TABLE activity(
    user_id INT,
    login_count INT
);

Step 3: SQL Query Tasks

1. Retrieve each user’s name and order amount using an INNER JOIN between users and orders.
Ans - SELECT users.name, orders.amount FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id;

2. Retrieve user name and total order amount for each user.
Ans - SELECT users.name, SUM(orders.amount) AS total_order_amount FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id
      GROUP BY users.name;

3. Retrieve all users and their login counts, including users with no recorded activity.
Ans - SELECT users.name, activity.login_count FROM users
      LEFT JOIN activity
      ON users.user_id = activity.user_id;

4. Retrieve users who have never placed an order.
Ans - SELECT users.name FROM users
      LEFT JOIN orders
      ON users.user_id = orders.user_id
      WHERE orders.order_id IS NULL;

5. Combine all three tables to return:
    a. user_id
    b. name
    c. total_order_amount
    d. login_count
Ans - SELECT users.user_id, users.name, SUM(orders.amount) AS total_order_amount, activity.login_count
      FROM users
      LEFT JOIN orders
      ON users.user_id = orders.user_id
      LEFT JOIN activity
      ON users.user_id = activity.user_id
      GROUP BY users.user_id, users.name, activity.login_count;

6. Retrieve users who placed orders greater than 400.
Ans - SELECT DISTINCT users.name FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id
      WHERE orders.amount > 400;

7. Retrieve the top 3 users by total order amount.
Ans - SELECT users.name, SUM(orders.amount) AS total_order_amount FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id
      GROUP BY users.name, users.user_id
      ORDER BY total_order_amount DESC
      LIMIT 3;

8. Retrieve users sorted by:
    a. total order amount (descending)
    b. login count (descending)
Ans - SELECT users.name, SUM(orders.amount) AS total_order_amount, activity.login_count
      FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id
      INNER JOIN activity
      ON users.user_id = activity.user_id
      GROUP BY users.name, users.user_id, activity.login_count
      ORDER BY total_order_amount DESC, login_count DESC;

9. Return the average order amount per user.
Ans - SELECT users.name, AVG(orders.amount) AS avg_order_amount FROM users
      INNER JOIN orders
      ON users.user_id = orders.user_id
      GROUP BY users.name, users.user_id;
    
10. Write one combined query that:
    a. Selects user_id, name, total_order_amount
    b. Includes login counts
    c. Sorts by total order amount (descending)
    d. Returns top 3 users
Ans - SELECT users.name, users.user_id, SUM(orders.amount) AS total_order_amount, activity.login_count
      FROM users
      LEFT JOIN orders
      ON users.user_id = orders.user_id
      LEFT JOIN activity
      ON users.user_id = activity.user_id
      GROUP BY users.name, users.user_id, activity.login_count
      ORDER BY total_order_amount DESC
      LIMIT 3;
